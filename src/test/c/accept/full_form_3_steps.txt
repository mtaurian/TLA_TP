@Step s1 {
	#Title "Step 1: Data"
	#Description "Fill your personal Data"
	@GetAwayCar {
		when howOld isLowerThan 13 goto END
		when howOld isGreaterThan 18 or howOld isEmpty goto s3 
	}
	@Section {
		#Title "First Section of s1"
		#Description "First Section Desc"
		@ShowIf isOlderThan18 { howOld isGreaterThan 18 }

		@Question name {
			#Type text
			#Title "Name"
			#PlaceHolder "Enter your name"
			#Default "."
			#Required
			#Help "your name goes here"
			@Glitch {
				@Error {
					#Message "Too long"
					@ShowIf { name isLongerThan 20 }
				}
				@Error {
					#Message "Too short"
					@ShowIf { name isShorterThan 1 or name isEmpty }
				}
			}
		}

		@Question howOld {
			#Type numeric
			#Title "How old are you"
			#Required		
		}

		@Question favoriteBeer {
			#Type select text
			#Title "Whats your favorite beer"
			#Options ["Brhama", "Patagonia", "Heineken" @ShowIf { howOld isGreaterThan 21 }]
			@ShowIf(isOlderThan18)
		}		
	}
	@Section {
		#Title Second Section of s1"
		#Description "Second Section Desc"

		@Question brhama {
			#Type text
			#Title "Whats your favorite brhama?"
			@ShowIf { favoriteBeer equals "brhama" }
		}
		
		@Question patagonia {
			#Type text
			#Title "Whats your favorite patagonia?"
			@ShowIf { favoriteBeer equals "patagonia" }
		}
		
		@ShowIf {
			(favoriteBeer equals "brhama" or favoriteBeer equals "Patagonia") and howOld isGreaterThan 18
		}
	}
	
}
